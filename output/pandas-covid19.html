import pandas as pd
import matplotlib.pyplot as plt
%matplotlib inline

file = "./csse_covid_19_data/csse_covid_19_time_series/time_series_covid19_confirmed_global.csv"
data = pd.read_csv(file)
data = data.drop(['Province/State', "Lat", "Long"], axis=1)
data = data.transpose()
data.columns = data.iloc[0]; data = data[1:]
data = data.groupby(data.columns, axis=1).sum().astype("int")
dailydf = data.diff(periods=7)

countries = ['US', 'Spain', 'France', 'Italy', 'Germany', "India", 'Australia']
fig = plt.figure()
ax = fig.add_subplot(1,1,1)
fig.set_figwidth(12)
fig.set_figheight(8)
ax.set_xlabel("Total Confirmed cases")
ax.set_ylabel("Weekly Confirmed cases")


for c in countries:
    x = dailydf[c]
    y = data[c]
    ax.loglog(y[:],x[:], label=f"{c}")
ax.legend()
ax.set_xlim(100, None)
ax.set_ylim((100, None))
